75const char *
76snmp_api_error(int err)
77{
78 int foo = (err * -1);
79 if ((foo < SNMPERR_GENERR) ||
80 (foo > SNMPERR_LAST))
81 foo = 0;
82
83 return (api_errors[foo]);
84}
