280bool
281Http::IsReforwardableStatus(const StatusCode s)
282{
283 switch (s) {
284
285 case scBadGateway:
286 case scGatewayTimeout:
287 return true;
288
289 case scForbidden:
290 case scInternalServerError:
291 case scNotImplemented:
292 case scServiceUnavailable:

294
295 default:
296 return false;
297 }
298
299 /* NOTREACHED */
300}
