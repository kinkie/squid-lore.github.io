124Mgr::InfoAction::respond(const Request& request)
125{
126 debugs(16, 5, MYNAME);
127 Ipc::ImportFdIntoComm(request.conn, SOCK_STREAM, IPPROTO_TCP, Ipc::fdnHttpSocket);
128 Must(Comm::IsConnOpen(request.conn));
