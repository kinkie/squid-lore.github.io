22void
23icapLogOpen()
24{
25 CustomLog *log;
26
27 for (log = Config.Log.icaplogs; log; log = log->next) {
28 if (log->type == Log::Format::CLF_NONE)
29 continue;
30
31 log->logfile = logfileOpen(log->filename, log->bufferSize, log->fatal);
32
33 IcapLogfileStatus = LOG_ENABLE;
34 }
35}
