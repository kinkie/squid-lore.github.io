20void
21CommIO::Initialize()
22{
23 if (CommIO::Initialized)
24 return;
25
26 /* Initialize done pipe signal */
27 int DonePipe[2];
28 if (pipe(DonePipe)) {}
29 DoneFD = DonePipe[1];
30 DoneReadFD = DonePipe[0];
31 fd_open(DoneReadFD, FD_PIPE, "async-io completion event: main");
32 fd_open(DoneFD, FD_PIPE, "async-io completion event: threads");
33 commSetNonBlocking(DoneReadFD);
34 commSetNonBlocking(DoneFD);
35 Comm::SetSelect(DoneReadFD, COMM_SELECT_READ, NULLFDHandler, nullptr, 0);
36 Initialized = true;
37}

39void
40CommIO::NotifyIOClose()
41{
42 /* Close done pipe signal */
43 FlushPipe();
44 xclose(DoneFD);
45 xclose(DoneReadFD);
46 fd_close(DoneFD);
47 fd_close(DoneReadFD);
48 Initialized = false;
49}
