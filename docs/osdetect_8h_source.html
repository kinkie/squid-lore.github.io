18/*
19 * Define the _SQUID_TYPE_ based on a guess of the OS.

40#elif defined(_AIX) /* AIX */

46#elif defined(__FreeBSD__) /* FreeBSD */

49#elif defined(__FreeBSD_kernel__) /* GNU/kFreeBSD */

76#elif defined(__QNX__)
