255static void

257{
258 if (!IamWorkerProcess())
259 return;
260
261 if (Config.Port.snmp <= 0)
262 return;
263
264 snmpIncomingConn = new Comm::Connection;
265 snmpIncomingConn->local = Config.Addrs.snmp_incoming;
266 snmpIncomingConn->local.port(Config.Port.snmp);
267
268 if (!Ip::EnableIpv6 && !snmpIncomingConn->local.setIPv4()) {
269 debugs(49, DBG_CRITICAL, "ERROR: IPv6 is disabled. " << snmpIncomingConn->local << " is not an IPv4 address.");
270 fatal("SNMP port cannot be opened.");

387/*
388 * Turn SNMP packet into a PDU, check available ACL's
389 */
390static void
391snmpDecodePacket(SnmpRequest * rq)
392{
393 struct snmp_pdu *PDU;
394 u_char *Community;
395 u_char *buf = rq->buf;
396 int len = rq->len;
397
398 if (!Config.accessList.snmp) {
399 debugs(49, DBG_IMPORTANT, "WARNING: snmp_access not configured. agent query DENIED from : " << rq->from);
400 return;
401 }
402
403 debugs(49, 5, "Called.");
404 PDU = snmp_pdu_create(0);
405 /* Always answer on SNMPv1 */
406 rq->session.Version = SNMP_VERSION_1;
407 Community = snmp_parse(&rq->session, PDU, buf, len);
408
409 /* Check if we have explicit permission to access SNMP data.
410 * default (set above) is to deny all */
411 if (Community) {
412 ACLFilledChecklist checklist(Config.accessList.snmp, nullptr);
413 checklist.src_addr = rq->from;
414 checklist.snmp_community = (char *) Community;
415
416 if (checklist.fastCheck().allowed() && (snmp_coexist_V2toV1(PDU))) {
417 rq->community = Community;
418 rq->PDU = PDU;
419 debugs(49, 5, "snmpAgentParse: reqid=[" << PDU->reqid << "]");
420 snmpConstructReponse(rq);
421 } else {
422 debugs(49, DBG_IMPORTANT, "WARNING: SNMP agent query DENIED from : " << rq->from);
423 snmp_free_pdu(PDU);
424 }
425 xfree(Community);
426
427 } else {
428 debugs(49, DBG_IMPORTANT, "WARNING: Failed SNMP agent query from : " << rq->from);
429 snmp_free_pdu(PDU);
430 }
431}

433/*
434 * Packet OK, ACL Check OK, Create response.
435 */
436static void
437snmpConstructReponse(SnmpRequest * rq)
438{
439
440 struct snmp_pdu *RespPDU;
441
442 debugs(49, 5, "snmpConstructReponse: Called.");
443
444 if (UsingSmp() && IamWorkerProcess()) {
445 AsyncJob::Start(new Snmp::Forwarder(static_cast<Snmp::Pdu&>(*rq->PDU),
446 static_cast<Snmp::Session&>(rq->session), rq->sock, rq->from));
447 snmp_free_pdu(rq->PDU);
448 return;
449 }
450
451 RespPDU = snmpAgentResponse(rq->PDU);
452 snmp_free_pdu(rq->PDU);

456 comm_udp_sendto(rq->sock, rq->from, rq->outbuf, rq->outlen);
457 else
458 debugs(49, DBG_IMPORTANT, "ERROR: Failed to encode a response to SNMP agent query from " << rq->from);
459 snmp_free_pdu(RespPDU);
460 }

463/*
464 * Decide how to respond to the request, construct a response and
465 * return the response to the requester.
466 */
467
468struct snmp_pdu *
469snmpAgentResponse(struct snmp_pdu *PDU) {
470
471 struct snmp_pdu *Answer = nullptr;
472
473 debugs(49, 5, "snmpAgentResponse: Called.");
474
475 if ((Answer = snmp_pdu_create(SNMP_PDU_RESPONSE))) {
476 Answer->reqid = PDU->reqid;
477 Answer->errindex = 0;
478
479 if (PDU->command == SNMP_PDU_GET || PDU->command == SNMP_PDU_GETNEXT) {
480 /* Indirect way */
481 int get_next = (PDU->command == SNMP_PDU_GETNEXT);
482 variable_list *VarPtr_;
483 variable_list **RespVars = &(Answer->variables);
484 oid_ParseFn *ParseFn;
485 int index = 0;
486 /* Loop through all variables */
487
488 for (VarPtr_ = PDU->variables; VarPtr_; VarPtr_ = VarPtr_->next_variable) {
489 variable_list *VarPtr = VarPtr_;
490 variable_list *VarNew = nullptr;
491 oid *NextOidName = nullptr;
492 snint NextOidNameLen = 0;
493
494 ++index;
495
496 if (get_next)
497 ParseFn = snmpTreeNext(VarPtr->name, VarPtr->name_length, &NextOidName, &NextOidNameLen);
498 else
499 ParseFn = snmpTreeGet(VarPtr->name, VarPtr->name_length);
500
501 if (ParseFn == nullptr) {
502 Answer->errstat = SNMP_ERR_NOSUCHNAME;
503 debugs(49, 5, "snmpAgentResponse: No such oid. ");
504 } else {
505 if (get_next) {
506 VarPtr = snmp_var_new(NextOidName, NextOidNameLen);
507 xfree(NextOidName);
508 }
509
510 int * errstatTmp = &(Answer->errstat);
511
512 VarNew = (*ParseFn) (VarPtr, (snint *) errstatTmp);
513
514 if (get_next)
515 snmp_var_free(VarPtr);
516 }
517
518 if ((Answer->errstat != SNMP_ERR_NOERROR) || (VarNew == nullptr)) {
519 Answer->errindex = index;
520 debugs(49, 5, "snmpAgentResponse: error.");
521
522 if (VarNew)
523 snmp_var_free(VarNew);
524
525 while ((VarPtr = Answer->variables) != nullptr) {
526 Answer->variables = VarPtr->next_variable;
527 snmp_var_free(VarPtr);
528 }
529
530 /* Steal the original PDU list of variables for the error response */
531 Answer->variables = PDU->variables;
532
533 PDU->variables = nullptr;
534
535 return (Answer);
536 }
537
538 /* No error. Insert this var at the end, and move on to the next.
539 */
540 *RespVars = VarNew;
541
542 RespVars = &(VarNew->next_variable);
543 }
544 }
545 }
546
547 return (Answer);
548}

604static oid_ParseFn *
605snmpTreeNext(oid * Current, snint CurrentLen, oid ** Next, snint * NextLen)
606{
607 oid_ParseFn *Fn = nullptr;
608 int count = 0;
609
610 debugs(49, 5, "snmpTreeNext: Called");
611
612 MemBuf tmp;
613 debugs(49, 6, "snmpTreeNext: Current : " << snmpDebugOid(Current, CurrentLen, tmp));
614
615 mib_tree_entry *mibTreeEntry = mib_tree_head;
616
617 if (mibTreeEntry && Current[count] == mibTreeEntry->name[count]) {
618 ++count;
619
620 while ((mibTreeEntry) && (count < CurrentLen) && (!mibTreeEntry->parsefunction)) {
621 mib_tree_entry *nextmibTreeEntry = snmpTreeEntry(Current[count], count, mibTreeEntry);
622
623 if (!nextmibTreeEntry)
624 break;
625 else
626 mibTreeEntry = nextmibTreeEntry;
627
628 ++count;
629 }
630 debugs(49, 5, "snmpTreeNext: Recursed down to requested object");
631 } else {
632 return nullptr;
633 }
634
635 if (mibTreeEntry == mib_tree_last)
636 return (Fn);
637
638 if ((mibTreeEntry) && (mibTreeEntry->parsefunction)) {
639 *NextLen = CurrentLen;
640 *Next = (*mibTreeEntry->instancefunction) (Current, NextLen, mibTreeEntry, &Fn);
641 if (*Next) {
642 debugs(49, 6, "snmpTreeNext: Next : " << snmpDebugOid(*Next, *NextLen, tmp));
643 return (Fn);
644 }
645 }
646
647 if ((mibTreeEntry) && (mibTreeEntry->parsefunction)) {
648 --count;
649 mib_tree_entry *nextoid = snmpTreeSiblingEntry(Current[count], count, mibTreeEntry->parent);
650 if (nextoid) {
651 debugs(49, 5, "snmpTreeNext: Next OID found for sibling" << nextoid );
652 mibTreeEntry = nextoid;
653 ++count;
654 } else {
655 debugs(49, 5, "snmpTreeNext: Attempting to recurse up for next object");
656
657 while (!nextoid) {
658 --count;
659
660 if (mibTreeEntry->parent->parent) {
661 nextoid = mibTreeEntry->parent;
662 mibTreeEntry = snmpTreeEntry(Current[count] + 1, count, nextoid->parent);
663
664 if (!mibTreeEntry) {
665 mibTreeEntry = nextoid;
666 nextoid = nullptr;
667 }
668 } else {
669 nextoid = mibTreeEntry;
670 mibTreeEntry = nullptr;
671 }
672 }
673 }
674 }
675 while ((mibTreeEntry) && (!mibTreeEntry->parsefunction)) {
676 mibTreeEntry = mibTreeEntry->leaves[0];
677 }
678
679 if (mibTreeEntry) {
680 *NextLen = mibTreeEntry->len;
681 *Next = (*mibTreeEntry->instancefunction) (mibTreeEntry->name, NextLen, mibTreeEntry, &Fn);
682 }
683
684 if (*Next) {
685 debugs(49, 6, "snmpTreeNext: Next : " << snmpDebugOid(*Next, *NextLen, tmp));
686 return (Fn);
687 } else
688 return nullptr;
689}

779{

784 int newshift = 0;
785

787 aux = client_entry(nullptr);

789 laddr = *aux;

805 *len += size ;
806 }
807 } else {
808 int shift = *len - current->len ; // i.e 4 or 16

812 laddr = *aux;

814 laddr.setAnyAddr();

827 *len = current->len + newshift ;
828 }
829 }
830

887mib_tree_entry *
888snmpLookupNodeStr(mib_tree_entry *root, const char *str)
889{
890 oid *name;
891 int namelen;
892 mib_tree_entry *e;
893
894 if (root)
895 e = root;
896 else
897 e = mib_tree_head;
898
899 if (! snmpCreateOidFromStr(str, &name, &namelen))
900 return nullptr;
901
902 /* I wish there were some kind of sensible existing tree traversal
903 * routine to use. I'll worry about that later */
904 if (namelen <= 1) {
905 xfree(name);
906 return e; /* XXX it should only be this? */
907 }
908
909 int i, r = 1;
910 while (r < namelen) {
911
912 /* Find the child node which matches this */
913 for (i = 0; i < e->children && e->leaves[i]->name[r] != name[r]; ++i) ; // seek-loop
914
915 /* Are we pointing to that node? */
916 if (i >= e->children)
917 break;
918 assert(e->leaves[i]->name[r] == name[r]);
919
920 /* Skip to that node! */
921 e = e->leaves[i];
922 ++r;
923 }
924
925 xfree(name);
926 return e;
927}

929bool
930snmpCreateOidFromStr(const char *str, oid **name, int *nl)
931{
932 char const *delim = ".";
933
934 *name = nullptr;
935 *nl = 0;
936 const char *s = str;
937
938 /* Parse the OID string into oid bits */
939 while (size_t len = strcspn(s, delim)) {
940 *name = (oid*)xrealloc(*name, sizeof(oid) * ((*nl) + 1));
941 (*name)[*nl] = atoi(s); // stops at the '.' delimiter
942 ++(*nl);
943 // exit with true when the last octet has been parsed
944 if (s[len] == '\0')
945 return true;
946 s += len+1;
947 }
948
949 // if we aborted before the lst octet was found, return false.
950 safe_free(*name);
951 return false;
952}

954/*
955 * Create an entry. Return a pointer to the newly created node, or NULL
956 * on failure.
957 */
958static mib_tree_entry *
959snmpAddNodeStr(const char *base_str, int o, oid_ParseFn * parsefunction, instance_Fn * instancefunction, AggrType aggrType)
960{
961 mib_tree_entry *m, *b;
962 oid *n;
963 int nl;
964 char s[1024];
965
966 /* Find base node */
967 b = snmpLookupNodeStr(mib_tree_head, base_str);
968 if (! b)
969 return nullptr;
970 debugs(49, 5, "snmpAddNodeStr: " << base_str << ": -> " << b);
971
972 /* Create OID string for new entry */
973 snprintf(s, 1024, "%s.%d", base_str, o);
974 if (! snmpCreateOidFromStr(s, &n, &nl))
975 return nullptr;
976
977 /* Create a node */
978 m = snmpAddNode(n, nl, parsefunction, instancefunction, aggrType, 0);
